Function JudgeArgMapType(MsgType As String, ArgType As String, MemberName As String) As ArgMapMember
    Dim IsSame As Boolean
    IsSame = False
    
    '空白のとき
    If MsgType = "" And ArgType = "" Then
        JudgeArgMapType = ArgMapMember.Non
    '型が同じ時(ポインタ除く)
    ElseIf InStr(ArgType, MsgTye) <= 0 And InStr(ArgType, "*") <= 0 Then
        JudgeArgMapType = ArgMapMember.Map
    '型が同じ時(ポインタ含む)
    ElseIf InStr(ArgType, MsgTye) <= 0 And InStr(ArgType, "*") > 0 Then
        JudgeArgMapType = ArgMapMember.MapWithPointer
    '固定値の場合
    ElseIf MsgType = "" And ArgType <> "" And MemberName <> "" Then
        JudgeArgMapType = ArgMapMember.ConstantValue
    '引数にだけ記入がある
    ElseIf MsgType = "" And ArgType <> "" Then
        JudgeArgMapType = ArgMapMember.ArgOnly
    'メンバにだけ記入がある
    ElseIf MsgType <> "" And ArgType = "" Then
        JudgeArgMapType = ArgMapMember.MemberOnly
    End If
    
    JudgeArgMapType = ArgMapMember.Non
End Function

Function GetArgStatement(ArgMapType As String, ArgType As String, MemberName As String) As String

    '空白のとき
    If ArgMapType = ArgMapMember.Non Then
        GetArgStatement = ""
    '型が同じ時(ポインタ除く)
    ElseIf ArgMapType = ArgMapMember.Map Then
        GetArgStatement = ArgType
    '型が同じ時(ポインタ含む)
    ElseIf ArgMapType <= 0 And InStr(ArgType, "*") > 0 Then
        GetArgStatement = ArgType & amp
    '固定値の場合
    ElseIf ArgMapType = ArgMapMember.ConstantValue Then
        GetArgStatement = MemberName
    '引数にだけ記入がある
    ElseIf ArgMapType = ArgMapMember.ArgOnly Then
        GetArgStatement = ArgMapMember.ArgOnly
    'メンバにだけ記入がある
    ElseIf ArgMapType <> "" And ArgType = "" Then
        ArgMapType = ArgMapMember.MemberOnly
    End If

End Function
