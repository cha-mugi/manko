Function JudgeArgMapType(MsgType As String, ArgType As String, MemberName As String) As ArgMapMember
    Dim IsSame As Boolean
    IsSame = False
    
    '空白のとき
    If MsgType = "" And ArgType = "" Then
        JudgeArgMapType = ArgMapMember.Non
    '型が同じ時(ポインタ除く)
    ElseIf InStr(ArgType, MsgTye) <= 0 And InStr(ArgType, "*") <= 0 Then
        JudgeArgMapType = ArgMapMember.Map
    '型が同じ時(ポインタ含む)
    ElseIf InStr(ArgType, MsgTye) <= 0 And InStr(ArgType, "*") > 0 Then
        JudgeArgMapType = ArgMapMember.MapWithPointer
    '固定値の場合
    ElseIf MsgType = "" And ArgType <> "" And MemberName <> "" Then
        JudgeArgMapType = ArgMapMember.ConstantValue
    '引数にだけ記入がある
    ElseIf MsgType = "" And ArgType <> "" Then
        JudgeArgMapType = ArgMapMember.ArgOnly
    'メンバにだけ記入がある
    ElseIf MsgType <> "" And ArgType = "" Then
        JudgeArgMapType = ArgMapMember.MemberOnly
    End If
    
    JudgeArgMapType = ArgMapMember.Non
End Function
